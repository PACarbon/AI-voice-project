# 开发板绘制

## 作业绘制法国国旗代码实现
```c
#include <stdio.h>
#include <sys/types.h>
#include <sys/stat.h>
#include <fcntl.h>
#include <unistd.h>

int main(int argc, char const *argv[])
{
    // 1.打开lcd屏幕
    int fd = open("/dev/fb0", O_RDWR);

    if (fd == -1)
    {
        perror("open /dev/fb0 failed!");
        // 打开失败退出主程序
        return -1;
    }

    // 2. 准备渲染在屏幕上数组
    int buf[480][800];

    // 3. 设置二维数组的各个像素的颜色
    /*
    for (int y = 0; y < 480; y++)
    {
        for (int x = 0; x < 800; x++)
        {

            if (x <= 266) // 法国国旗蓝色
            {
                buf[y][x] = 0xff3f48cc;
            }
            else if (x <= 2 * 266)
            {
                buf[y][x] = 0xffffffff;
            }
            else
            {
                buf[y][x] = 0xffb70000;
            }
        }
    }
    */
    // 优化
    for (int y = 0; y < 480; y++)
    {
        for (int x = 0; x < 800; x++)
        {
            int color = 0xffb70000; // 深红

            if (x <= 266) // 法国国旗蓝色
            {
                color = 0xff3f48cc;
            }
            else if (x <= 2 * 266) // 法国国旗白色
            {
                color = 0xffffffff;
            }

            buf[y][x] = color;
        }
    }

    // 将颜色数据写入到lcd屏幕中
    int ret = write(fd, buf, 800 * 480 * 4);
    if (ret == -1)
    {
        perror("write buf failed!");
    }
    
    // 关闭文件描述符
    close(fd);

    return 0;
}
```

代码完成后在Ubuntu中编译
```shell
arm-linux-gcc 绘制法国国旗实现.c -o a
```

电脑连接开发板，并在开发板的根目录下创建`demo`文件夹,并访问
```shell
cd /

mkdir demo

cd /demo
```

将编译后的文件传入到该目录内输入指令`rx 文件名`，将本地文件拖拽至SecureCRT，选择“发送Xmodem”
```shell
rx a
```

如果是传入的是新文件，在运行该程序前需要修改他的权限
```shell
chmod 777 文件名
```

启动 `./文件名`
